




CREATE DATABASE LEILAOAPP;
USE LEILAOAPP;
CREATE TABLE TABPRODUTO (
	CODPROD INT  AUTO_INCREMENT NOT NULL PRIMARY KEY,
	NOMEPROD VARCHAR(25) NOT NULL,
	VALORPROD DECIMAL(5,2) NOT NULL,
	QUANTPROD INT(3)
);
	
CREATE TABLE TABCLIENTE (

	IDCLI INT NOT NULL PRIMARY KEY,
	NOME_CLIENTE VARCHAR(25) NOT NULL,
	RG_CLIENTE VARCHAR(14) NOT NULL,
	CPF_CLIENTE VARCHAR(14) NOT NULL,
	DATANASC_CLIENTE DATE NOT NULL,
	SEXO_CLIENTE VARCHAR(1) CHECK (SEXO_CLIENTE IN ('M','H') NOT NULL),

	ENDERECO_CLIENTE VARCHAR(25) NOT NULL,
	ENDERECO_CLIENTE_NUMERO VARCHAR(9) NOT NULL,
	ENDERECO_CLIENTE_COMPLEMENTO VARCHAR(25),
	ENDERECO_CLIENTE_CEP VARCHAR(10) NOT NULL,
	ENDERECO_CLIENTE_CIDADE VARCHAR(25) NOT NULL,
	ENDERECO_CLIENTE_UF VARCHAR(2) NOT NULL,
	ENDERECO_CLIENTE_PAIS VARCHAR(15) NOT NULL,

	EMAIL_CLIENTE VARCHAR(25) NOT NULL,
	CELULAR_CLIENTE VARCHAR(15) NOT NULL,
	TELEFONE_CLIENTE VARCHAR(15),
	SENHA_CLIENTE VARCHAR(15) NOT NULL
);

CREATE TABLE TABFUNC (
	IDFUNC INT NOT NULL PRIMARY KEY,
	CPFFUNC VARCHAR(14) NOT NULL,
	NOMEFUNC VARCHAR(25) NOT NULL,
	DATAFUNC DATE NOT NULL,
	TELFUNC VARCHAR(14) NOT NULL,
	SALFUNC DECIMAL(8,2) NOT NULL
);

CREATE TABLE TABVENDA (
	IDVENDA INT(3) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	NUMVENDA INT(3) NOT NULL,
	DATAVENDA DATE NOT NULL,
	IDPROD INT(3) NOT NULL,
	VALORTOTALVENDA(50) VARCHAR NOT NULL,
	DESCONTOVENDA VARCHAR(50) NOT NULL,
	TOTALPAGO VARCHAR(50) NOT NULL,
	IDFUNC INT(3) NOT NULL,
	IT CLI INT(3)
);

CREATE TABLE TABPRODUTO (
    IDPRODUTO INT(3) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	NOME_PRODUTO VARCHAR(25) NOT NULL,
	DESCRICAO_PRODUTO VARCHAR (100),
	LANCE_PRODUTO VARCHAR (4) NOT NULL,
	STATUS (4) NOT NULL
);

CREATE TABLE TABFUNCIONARIO (	
	IDFUNCIONARIO INT(3) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	EMAIL_CLIENTE VARCHAR(25) NOT NULL,
	SENHA_CLIENTE VARCHAR(15) NOT NULL
);

CREATE TABLE TABVENDA (
	IDVENDA INT(3) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	IDPRODUTO INT(3) NOT NULL,
	IDCLIENTE INT(3) NOT NULL,
	NOME_PRODUTO VARCHAR(25) NOT NULL,
	DESCRICAO_PRODUTO VARCHAR (100),
	LANCE_INICIAL VARCHAR (4) NOT NULL,
	LANCE_FINAL VARCHAR (4) NOT NULL,
	EMAIL_CLIENTE VARCHAR(25) NOT NULL,
	SENHA_CLIENTE VARCHAR(15) NOT NULL	
);


STATUS = ATI, NAT, VEND
